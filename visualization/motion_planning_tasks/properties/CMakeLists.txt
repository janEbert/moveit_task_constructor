find_package(agni_tf_tools REQUIRED)

set(MOVEIT_LIB_NAME motion_planning_tasks_properties)

set(SOURCES
	property_factory.cpp
	frame_selector.cpp
	move_group_selector.cpp
	pose_stamped_selector.cpp
	twist_stamped_selector.cpp
)
add_library(${MOVEIT_LIB_NAME} SHARED ${SOURCES})

target_link_libraries(${MOVEIT_LIB_NAME}
	PUBLIC ${QT_LIBRARIES} ${catkin_LIBRARIES}
	PRIVATE ${agni_tf_tools_LIBRARIES}
)
target_include_directories(${MOVEIT_LIB_NAME}
	PUBLIC $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/..>
	PRIVATE ${catkin_INCLUDE_DIRS}
)

install(TARGETS ${MOVEIT_LIB_NAME}
	ARCHIVE DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
	LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION})
